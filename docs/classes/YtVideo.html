getId: ->
	match = @Url.match /^.*(youtu.be\/|v\/|u\/\w\/|embed\/|watch\?v=|\&v=)([^#\&\?]*).*/
	if match && match[2].length == 11
		match[2]
	else
		@emit 'error', 'Invalid YouTube link.'
		false
